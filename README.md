# Base64

## Install
`npm i --save @dwlib/base64`

## Usage
```javascript
// CJS
const Base64 = require('@dwlib/base64');
const Base64Decode = require('@dwlib/base64/decode');
const Base64DecodeInto = require('@dwlib/base64/decodeInto');
const Base64Encode = require('@dwlib/base64/encode');
const Base64IsValid = require('@dwlib/base64/isValid');
const Base64Normalize = require('@dwlib/base64/normalize');
const Base64Validate = require('@dwlib/base64/validate');
// ESM
import Base64 from '@dwlib/base64';
import Base64Decode from '@dwlib/base64/decode';
import Base64DecodeInto from '@dwlib/base64/decodeInto';
import Base64Encode from '@dwlib/base64/encode';
import Base64IsValid from '@dwlib/base64/isValid';
import Base64Normalize from '@dwlib/base64/normalize';
import Base64Validate from '@dwlib/base64/validate';
```
