# Base64

## Install
`npm i --save @dwlib/base64`

## Usage
```javascript
// CJS
const base64 = require('@dwlib/base64');
// ESM
import Base64 from '@dwlib/base64';
import * as base64 from '@dwlib/base64';
// Module Exports
const {
  Base64,
  encode,
  encodeToBytes,
  decode,
  decodeToBytes,
  encodeBytes,
  encodeBytesToString,
  decodeBytes,
  decodeBytesToString,
  encodeText,
  encodeTextToBytes,
  decodeText,
  decodeBytesToText,
  encodeInt,
  decodeInt,
  encodeBigInt,
  decodeBigInt
} = base64;
```
